## Features

- Provides text alignment options: left, right, center, or justify.

</PackageInfo>

## Installation

```bash
npm install @sewellstephens/plate-alignment
```

## Usage

```tsx
// ...
import { createLateEditor } from '@sewellstephens/plate-common/react';
import { AlignPlugin } from '@sewellstephens/plate-alignment/react';
import { ParagraphPlugin } from '@sewellstephens/plate-common/react';
import { HeadingPlugin } from '@sewellstephens/plate-heading/react';

const editor = createLateEditor({
  plugins: [
    HeadingPlugin,
    AlignPlugin.configure({
      inject: {
        targetPlugins: [
          ParagraphPlugin.key,
          HeadingPlugin.key,
        ],
      },
    }),
  ],
});
```
